lhh_residential_area_max_level = {
	add = {
		desc = "BUILDING_LEVEL_DEVELOPMENT"
		value = development
		multiply = 0.1
	}
	if = {
		limit = {
			location_rank ?= location_rank:rural_settlement
		}
		add = {
			desc = "BUILDING_LEVEL_IS_RURAL_SETTLEMENT"
			value = 5
		}
        add = {
            desc = "BUILDING_LEVEL_POPULATION"
            value = population
            multiply = 0.05
	    }
	}
    if = {
		limit = {
			location_rank ?= location_rank:town
		}
		add = {
			desc = "BUILDING_LEVEL_IS_TOWN"
			value = 5
		}
        add = {
            desc = "BUILDING_LEVEL_POPULATION"
            value = population
            multiply = 0.01
	    }
	}
	if = {
		limit = {
			location_rank ?= location_rank:city
		}
		add = {
			desc = "BUILDING_LEVEL_IS_CITY"
			value = 10
		}
        add = {
            desc = "BUILDING_LEVEL_POPULATION"
            value = population
            multiply = 0.025
	    }
	}
}

lhh_manol_form_max_level = {
	add = {
		desc = "BUILDING_LEVEL_POPULATION"
		value = population
		multiply = 0.005
	}
	if = {
		limit = {
			has_river = yes
		}
		add = {
			desc = "BUILDING_LEVEL_HAS_RIVER"
			value = 1
		}
	}
	if = {
		limit = {
			vegetation = farmland
		}
		add = {
			desc = "lhh_building_level_vegetation_is_farmland"
			value = 2
		}
	}
}
