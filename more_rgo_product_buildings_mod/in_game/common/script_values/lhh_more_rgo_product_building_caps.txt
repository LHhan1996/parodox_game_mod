lhh_iron_workshop_max_level = {
	add = {
		desc = "BUILDING_LEVEL_DEVELOPMENT"
		value = development
		multiply = 0.1
	}
	if = {
		limit = {
			raw_material ?= goods:iron
		}
		add = {
			desc = "BUILDING_LEVEL_RGO_MAX"
			value = max_rgo_workers
			multiply = 1.0
		}
	}
    if = {
		limit = {
			location_rank ?= location_rank:town
		}
		add = {
			desc = "BUILDING_LEVEL_IS_TOWN"
			value = 5
		}
		add = {
			desc = "BUILDING_LEVEL_POPULATION"
			value = population
			multiply = 0.02
		}
	}
	if = {
		limit = {
			location_rank ?= location_rank:city
		}
		add = {
			desc = "BUILDING_LEVEL_IS_CITY"
			value = 5
		}
		add = {
			desc = "BUILDING_LEVEL_POPULATION"
			value = population
			multiply = 0.05
		}
	}
}
